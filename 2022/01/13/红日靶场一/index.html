# 内网信息收集

# msf

# 当前用户
whoami
# 主机名
hostname
# 本机用户列表
net user
# 本地管理员组
net localgroup administrator
# 系统版本
systeminfo | findstr /B /C:"OS Name" /C:"OS Version"
# 系统架构
echo %PROCESSOR_ARCHITECTURE%
# 安装软件及版本
wmic product get name,version
# 进程与服务
tasklist
# IP查询，根据DNS判断是否在域环境
ipconfig/all
# 局域网内其他主机名
net view
# 查看域用户
net view /domain
# 查询计算机名 全名 用户名 系统版本 工作站 域 登录域
net config workstation
# 查看域控服务器
nslookup god.org
# 查看当前域下的所有用户 
net user /domain # 若发现能够执行，说明此台机器在域中 (若是此命令在显示域处显示 WORKGROUP，则不存在域，若是报错：发生系统错误 5，则存在域，但当前登录用户不是域用户)
# 获取域内用户信息
wmic useraccount get /all
# 查看域管理员的名字
net group "domain admins" /domain
# 查看所有域成员计算机名
net group "domain computers" /domain
# 查看域管理员
net group "domain admins" /domain
